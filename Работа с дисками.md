Почитать про LVM 

``` sh 
# создеем lvm раздел
sudo lvcreate -L 50G -n timeshift-lv ubuntu-vg
```

сделал новый ЛВМ


``` sh
# Форматируем в ext4
sudo mkfs.ext4 /dev/ubuntu-vg/timeshift-lv

# Проверяем
sudo blkid /dev/ubuntu-vg/timeshift-lv
```

тут история с Timeshift



``` sh 
# Создаем папку для монтирования
sudo mkdir /mnt/timeshift

# Монтируем раздел
sudo mount /dev/ubuntu-vg/timeshift-lv /mnt/timeshift

# Проверяем
df -h /mnt/timeshift
```


fff933c4-0b2b-48fc-b757-c583baa8fea9


``` sh
# Узнаем UUID раздела
TS_UUID=$(sudo blkid /dev/ubuntu-vg/timeshift-lv -s UUID -o value)
echo $TS_UUID
# Запишите этот UUID!

# Добавляем в fstab для автоматического монтирования при загрузке
echo "UUID=$TS_UUID /mnt/timeshift ext4 defaults 0 2" | sudo tee -a /etc/fstab

# Проверяем что всё работает
sudo umount /mnt/timeshift
sudo mount -a
df -h /mnt/timeshift  # должно снова показать раздел
```
